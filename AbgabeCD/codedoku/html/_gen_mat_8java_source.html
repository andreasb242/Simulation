00050     public Object evaluate(Node node, EvaluatorI pv) throws ParseException {
00051         Object sizeObj = pv.eval(node.jjtGetChild(0));
00052         int sizes[] = null;
00053         if (sizeObj instanceof Scaler) {
00054             sizes = new int[1];
00055             sizes[0] = ((Number) sizeObj).intValue();
00056         } else if (sizeObj instanceof MVector) {
00057             MVector vec = (MVector) sizeObj;
00058             int n = vec.getNumEles();
00059             sizes = new int[n];
00060             for (int i = 0; i < n; ++i)
00061                 sizes[i] = ((Number) vec.getEle(i)).intValue();
00062         } else if (sizeObj instanceof MatrixValueI)
00063             throw new ParseException("GenMat: First arg should be scaler or vector");
00064         else {
00065             sizes = new int[1];
00066             sizes[0] = ((Number) sizeObj).intValue();
00067         }
00068 
00069         // Create result object
00070         Dimensions dim = Dimensions.valueOf(sizes);
00071         MatrixValueI res = Tensor.getInstance(dim);
00072 
00073         if (node.jjtGetNumChildren() == 2) {
00074             // no need to set variables
00075             for (int i = 0; i < res.getNumEles(); ++i) {
00076                 Object val = pv.eval(node.jjtGetChild(1));
00077                 res.setEle(i, val);
00078             }
00079             return res;
00080         }
00081 
00082         // Need to set variables
00083 
00084         Variable vars[] = VMap.getVars(node.jjtGetChild(2));
00085         if (vars.length != sizes.length)
00086             throw new ParseException("GenMat: number of variables must match number of dimensions");
00087         if (vars.length == 1) {
00088             for (int i = 0; i < sizes[0]; ++i) {
00089                 vars[0].setValue(new Integer(i + 1));
00090                 Object val = pv.eval(node.jjtGetChild(1));
00091                 res.setEle(i, val);
00092             }
00093             return res;
00094         } else if (vars.length == 2) {
00095             Matrix mat = (Matrix) res;
00096             for (int i = 0; i < sizes[0]; ++i) {
00097                 vars[0].setValue(new Integer(i + 1));
00098                 for (int j = 0; j < sizes[1]; ++j) {
00099                     vars[1].setValue(new Integer(j + 1));
00100                     Object val = pv.eval(node.jjtGetChild(1));
00101                     mat.setEle(i, j, val);
00102                 }
00103             }
