00038     public MatrixValueI evaluate(MatrixNodeI node, MatrixEvaluator visitor, MatrixJep jep) throws ParseException {
00039         int nChild = node.jjtGetNumChildren();
00040         if (nChild < 3)
00041             throw new ParseException("Map must have three or more arguments");
00042 
00043         // First find the variables
00044         Variable vars[] = getVars(node.jjtGetChild(1));
00045 
00046         if (nChild != vars.length + 2)
00047             throw new ParseException("Map: number of arguments should match number of variables + 2");
00048 
00049         // Now evaluate third and subsequent arguments
00050         MatrixValueI inputs[] = new MatrixValueI[nChild - 2];
00051         Dimensions dim = null;
00052         for (int i = 0; i < nChild - 2; ++i) {
00053             Object out = node.jjtGetChild(i + 2).jjtAccept(visitor, null);
00054             if (out instanceof MatrixValueI) {
00055                 inputs[i] = (MatrixValueI) out;
00056                 if (i == 0)
00057                     dim = inputs[0].getDim();
00058                 else {
00059                     if (!dim.equals(inputs[i].getDim()))
00060                         throw new ParseException("Map: dimensions of third and subsequent arguments must match");
00061                 }
00062             } else
00063                 throw new ParseException("Map: third and following arguments should be vectos or matricies");
00064         }
00065 
00066         // Now evaluate the function for each element
00067         MatrixValueI res = node.getMValue();
00068         for (int i = 0; i < dim.numEles(); ++i) {
00069             for (int j = 0; j < vars.length; ++j) {
00070                 ((MatrixVariableI) vars[j]).getMValue().setEle(0, inputs[j].getEle(i));
00071                 vars[j].setValidValue(true);
00072             }
00073             Scaler val = (Scaler) node.jjtGetChild(0).jjtAccept(visitor, null);
00074             res.setEle(i, val.getEle(0));
00075         }
00076         return res;
