00079         MatrixNodeI children[] = new MatrixNodeI[arguments.length];
00080         for (int i = 0; i < arguments.length; ++i)
00081             children[i] = (MatrixNodeI) arguments[i];
00082 
00083         if (pfmc instanceof BinaryOperatorI) {
00084             if (children.length != 2)
00085                 throw new ParseException("Operator " + name + " must have two elements, it has " + children.length);
00086             BinaryOperatorI bin = (BinaryOperatorI) pfmc;
00087             Dimensions dim = bin.calcDim(children[0].getDim(), children[1].getDim());
00088             return dim;
00089         } else if (pfmc instanceof UnaryOperatorI) {
00090             if (children.length != 1)
00091                 throw new ParseException("Operator " + name + " must have one elements, it has " + children.length);
00092             UnaryOperatorI uni = (UnaryOperatorI) pfmc;
00093             Dimensions dim = uni.calcDim(children[0].getDim());
00094             return dim;
00095         } else if (pfmc instanceof NaryOperatorI) {
00096             Dimensions dims[] = new Dimensions[children.length];
00097             for (int i = 0; i < children.length; ++i)
00098                 dims[i] = children[i].getDim();
00099             // if(arguments.length!=1) throw new
00100             // ParseException("Operator "+op.getName()+" must have one elements, it has "+arguments.length);
00101             NaryOperatorI uni = (NaryOperatorI) pfmc;
00102             Dimensions dim = uni.calcDim(dims);
00103             return dim;
00104         } else {
00105             return Dimensions.ONE;
00106             // System.out.println("Warning: assuming 1 for dimensions of "+node.getName());
00107         }
