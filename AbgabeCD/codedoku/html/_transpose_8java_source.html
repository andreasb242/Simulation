00032     public MatrixValueI calcValue(MatrixValueI res, MatrixValueI lhs) throws ParseException {
00033         if (!(res instanceof Matrix))
00034             throw new ParseException("transpose: result must be a matrix");
00035         if (!(lhs instanceof Matrix))
00036             throw new ParseException("transpose: argument must be a matrix");
00037         Matrix resmat = (Matrix) res;
00038         Matrix mat = (Matrix) lhs;
00039         if (resmat.getNumCols() != mat.getNumRows() || resmat.getNumRows() != mat.getNumCols())
00040             throw new ParseException("transpose, dimension of result is wrong res " + resmat.getDim() + " arg " + mat.getDim());
00041         for (int i = 0; i < resmat.getNumRows(); ++i) {
00042             for (int j = 0; j < resmat.getNumCols(); ++j)
00043                 resmat.setEle(i, j, mat.getEle(j, i));
00044         }
00045         return res;
00046     }
