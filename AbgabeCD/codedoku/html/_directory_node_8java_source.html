00165     private ArrayList<File> getFiles(JFileChooser chooser) {
00166         // fixed bug #97124
00167         ArrayList<File> list = new ArrayList<File>();
00168 
00169         // Fix for IZ#116859 [60cat] Node update bug in the "open project" panel
00170         // while deleting directories
00171         if (directory == null || !directory.exists()) {
00172             return list;
00173         }
00174 
00175         File[] files = chooser.getFileSystemView().getFiles(directory, chooser.isFileHidingEnabled());
00176         int mode = chooser.getFileSelectionMode();
00177         if (mode == JFileChooser.DIRECTORIES_ONLY) {
00178             for (int i = 0; i < files.length; i++) {
00179                 File child = files[i];
00180                 if (child.isDirectory()) {
00181                     list.add(child);
00182                 }
00183             }
00184             Collections.sort(list, FILE_NAME_COMPARATOR);
00185         } else if (mode == JFileChooser.FILES_AND_DIRECTORIES || mode == JFileChooser.FILES_ONLY) {
00186             ArrayList<File> dirList = new ArrayList<File>();
00187             ArrayList<File> fileList = new ArrayList<File>();
00188             for (int i = 0; i < files.length; i++) {
00189                 File child = files[i];
00190                 if (child.isDirectory()) {
00191                     dirList.add(child);
00192                 } else {
00193                     fileList.add(child);
00194                 }
00195             }
00196 
00197             Collections.sort(dirList, FILE_NAME_COMPARATOR);
00198             Collections.sort(fileList, FILE_NAME_COMPARATOR);
00199 
00200             list.addAll(dirList);
00201             list.addAll(fileList);
00202         }
00203 
00204         return list;
00205     }
