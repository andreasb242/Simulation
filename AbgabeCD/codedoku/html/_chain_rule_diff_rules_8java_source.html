00055     public Node differentiate(ASTFunNode node, String var, Node[] children, Node[] dchildren, DJep djep) throws ParseException {
00056         XOperatorSet opSet = (XOperatorSet) djep.getOperatorSet();
00057         NodeFactory nf = djep.getNodeFactory();
00058         FunctionTable funTab = djep.getFunctionTable();
00059 
00060         int nRules = rules.length;
00061         if (nRules != children.length)
00062             throw new ParseException("Error differentiating " + name + " number of rules " + nRules + " != number of arguments " + children.length);
00063 
00064         if (nRules == 1) {
00065             // df(g(x))/dx -> f'(g(x)) * g'(x)
00066             Node fprime = djep.deepCopy(rules[0]);
00067             Node g = children[0];
00068             Node fprimesub = djep.substitute(fprime, "x", g);
00069             Node gprime = dchildren[0];
00070             return nf.buildOperatorNode(opSet.getMultiply(), fprimesub, gprime);
00071         } else if (nRules == 2) {
00072             // df(g(x),h(x))/dx -> df/dg * dg/dx + df/dh * dh/dx
00073             Node df_dg = djep.deepCopy(rules[0]);
00074             Node df_dh = djep.deepCopy(rules[1]);
00075             Node gprime = dchildren[0];
00076             Node hprime = dchildren[1];
00077             df_dg = djep.substitute(df_dg, new String[] { "x", "y" }, children);
00078             df_dh = djep.substitute(df_dh, new String[] { "x", "y" }, children);
00079 
00080             return nf.buildOperatorNode(opSet.getAdd(), nf.buildOperatorNode(opSet.getMultiply(), df_dg, gprime), nf.buildOperatorNode(opSet.getMultiply(),
00081                     df_dh, hprime));
00082         } else if (nRules < 1) {
00083             throw new ParseException("Error differentiating " + name + " zero differention rules!");
00084         } else {
00085             String names[] = new String[nRules];
00086             Node[] df_dg = new Node[nRules];
00087             Node[] products = new Node[nRules];
00088             for (int i = 0; i < nRules; ++i) {
00089                 df_dg[i] = djep.deepCopy(rules[i]);
00090                 names[i] = "x" + i;
00091             }
00092             for (int i = 0; i < nRules; ++i) {
00093                 df_dg[i] = djep.substitute(df_dg[i], names, children);
00094                 products[i] = nf.buildOperatorNode(opSet.getMultiply(), df_dg[i], dchildren[i]);
00095             }
00096             return nf.buildFunctionNode("sum", funTab.get("sum"), products);
00097         }
00098     }
