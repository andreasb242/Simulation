00067     static final String USE_SHELL_FOLDER = "FileChooser.useShellFolder";
00068     static final String NB_USE_SHELL_FOLDER = "nb.FileChooser.useShellFolder";
00069     static final String START_TIME = "start.time";

00073     public static ComponentUI createUI(JComponent c) {
00074         JFileChooser fc = (JFileChooser) c;
00075         
00076         // #109703 - don't use shell folder on JDK versions interval <1.6.0_02,
00077         // 1.6.0_10>,
00078         // it's terribly slow on Windows due to JDK bug
00079         if (OS.isWindows()) {
00080             if (System.getProperty(NB_USE_SHELL_FOLDER) != null) {
00081                 fc.putClientProperty(USE_SHELL_FOLDER, Boolean.getBoolean(NB_USE_SHELL_FOLDER));
00082             } else {
00083                 String jv = System.getProperty("java.version");
00084                 jv = jv.split("-", 2)[0];
00085                 if ("1.6.0_02".compareToIgnoreCase(jv) <= 0 && "1.6.0_10".compareToIgnoreCase(jv) >= 0) {
00086                     if (!Boolean.TRUE.equals(fc.getClientProperty(USE_SHELL_FOLDER))) {
00087                         fc.putClientProperty(USE_SHELL_FOLDER, Boolean.FALSE);
00088                     }
00089                 }
00090             }
00091         }
00092 
00093         // mark start time, just once during init (code can be run multiple
00094         // times because of property listenign below)
00095         if (fc.getClientProperty(START_TIME) == null) {
00096             fc.putClientProperty(START_TIME, Long.valueOf(System.currentTimeMillis()));
00097         }
00098 
00099         Class<?> chooser = getCurChooser(fc);
00100         ComponentUI compUI;
00101         try {
00102             Method createUIMethod = chooser.getMethod("createUI", JComponent.class);
00103             compUI = (ComponentUI) createUIMethod.invoke(null, fc);
00104         } catch (Exception exc) {
00105             Logger.getLogger(DelegatingChooserUI.class.getName()).log(Level.FINE, "Could not instantiate custom chooser, fallbacking to Metal", exc);
00106             compUI = MetalFileChooserUI.createUI(c);
00107         }
00108 
00109         // listen to sel mode changes and select correct chooser by invoking
00110         // filechooser.updateUI() which triggers this createUI again
00111         if (firstTime) {
00112             fc.addPropertyChangeListener(JFileChooser.FILE_SELECTION_MODE_CHANGED_PROPERTY, new PropertyChangeListener() {
00113                 public @Override
00114                 void propertyChange(PropertyChangeEvent evt) {
00115                     JFileChooser fileChooser = (JFileChooser) evt.getSource();
00116                     fileChooser.updateUI();
00117                 }
00118             });
00119         }
00120 
00121         return compUI;
00122     }
00123 
00128     private static Class<?> getCurChooser(JFileChooser fc) {
00129         if (fc.getFileSelectionMode() == JFileChooser.DIRECTORIES_ONLY) {
00130             return DirectoryChooserUI.class;
00131         }
00132         return FolderChooserUi.getOrigChooser();
00133     }
