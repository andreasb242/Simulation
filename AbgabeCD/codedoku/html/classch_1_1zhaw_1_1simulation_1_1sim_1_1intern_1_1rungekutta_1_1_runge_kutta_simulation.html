Runge-Kutta implementation

 loop {
 		i := 0;
 		k1 := hf(x[i], y[i]);
 		k2 := hf(x[i] + h / 2, y[i] + k1 / 2);
 		k3 := hf(x[i] + h / 2, y[i] + k2 / 2);
 		k4 := hf(x[i] + h, y[i] + k3);
 		i := i + 1;
 		x[i] := x[i-1] + h;
 		y[i] := y[i-1] + (k1 + 2k2 + 2k3 + k4) / 6;
 }
