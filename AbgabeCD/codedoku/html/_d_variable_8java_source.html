00131     private PartialDerivative findDerivativeSorted(String derivnames[], DJep jep) throws ParseException {
00132         if (getEquation() == null)
00133             return null;
00134 
00135         if (derivnames == null)
00136             throw new ParseException("findDerivativeSorted: Null array of names");
00137         PartialDerivative res = getDerivativeSorted(derivnames);
00138         if (res != null)
00139             return res;
00140 
00141         // Deriv not found. Calculate from lower derivative
00142         int origlen = derivnames.length;
00143         Node lowereqn; // equation for lower derivative (or root equation)
00144         if (origlen < 1)
00145             throw new ParseException("findDerivativeSorted: Empty Array of names");
00146         else if (origlen == 1)
00147             lowereqn = getEquation();
00148         else {
00149             String newnames[] = new String[origlen - 1];
00150             for (int i = 0; i < origlen - 1; ++i)
00151                 newnames[i] = derivnames[i];
00152             lowereqn = findDerivativeSorted(newnames, jep).getEquation();
00153         }
00154         if (lowereqn == null) {
00155             return null;
00156         }
00157         Node deriv = jep.differentiate(lowereqn, derivnames[origlen - 1]);
00158         Node simp = jep.simplify(deriv);
00159         res = createDerivative(derivnames, simp);
00160         setDerivative(derivnames, res);
00161         return res;
00162     }
