00041     public void set(EvaluatorI pv, Node node, Object value) throws ParseException {
00042         Node lhs = node.jjtGetChild(0);
00043         if (!(lhs instanceof ASTVarNode))
00044             throw new ParseException("ArrayAccess: lhs of operator must be a Variable");
00045         Variable var = ((ASTVarNode) lhs).getVar();
00046         MatrixValueI varVal = (MatrixValueI) var.getValue();
00047         Object indicies = pv.eval(node.jjtGetChild(1));
00048         if (varVal instanceof MVector) {
00049             if (indicies instanceof Number) {
00050                 ((MVector) varVal).setEle(((Number) indicies).intValue() - 1, value);
00051                 return;
00052             } else if (indicies instanceof MVector) {
00053                 MVector vec = (MVector) indicies;
00054                 if (vec.getDim().equals(Dimensions.ONE)) {
00055                     int d1 = ((Number) vec.getEle(0)).intValue();
00056                     if (d1 < 1 || d1 > ((MVector) varVal).getNumEles())
00057                         throw new ParseException("ArrayAccess: array index " + d1 + " out of range 1.." + varVal.getDim());
00058                     ((MVector) varVal).setEle(d1 - 1, value);
00059                     return;
00060                 }
00061             }
00062             throw new ParseException("ArrayAccess: Bad second argument expecting a double " + indicies.toString());
00063         } else if (varVal instanceof Matrix) {
00064             if (indicies instanceof MVector) {
00065                 MVector vec = (MVector) indicies;
00066                 if (vec.getDim().equals(Dimensions.TWO)) {
00067                     int d1 = ((Number) vec.getEle(0)).intValue();
00068                     int d2 = ((Number) vec.getEle(1)).intValue();
00069                     if (d1 < 1 || d1 > ((Matrix) varVal).getNumRows() || d2 < 1 || d2 > ((Matrix) varVal).getNumCols())
00070                         throw new ParseException("ArrayAccess: array indices " + d1 + ", " + d2 + " out of range 1.." + varVal.getDim());
00071                     ((Matrix) varVal).setEle(d1 - 1, d2 - 1, value);
00072                     return;
00073                 }
00074             } else
00075                 throw new ParseException("ArrayAccess:Bad second argument, expecting [i,j] " + indicies.toString());
00076         } else if (indicies instanceof Tensor) {
00077             throw new ParseException("ArrayAccess: Sorry don't know how to set an elements for a tensor");
00078         }
00079         throw new ParseException("ArrayAccess: requires a vector matrix or tensor for first argument it has " + varVal.toString());
