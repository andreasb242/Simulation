00370     private boolean isInSelection(int x, int y, int w, int h) {
00371         Rectangle selectionRange = getSelectionRange();
00372 
00373         int sX = (int) selectionRange.getX();
00374         int sY = (int) selectionRange.getY();
00375         int sW = (int) selectionRange.getWidth();
00376         int sH = (int) selectionRange.getHeight();
00377 
00378         int hw = w / 2;
00379         int hh = h / 2;
00380 
00381         int ex = x + w;
00382         int ey = y + h;
00383 
00384         // obere bzw. linke Hälfe selektiert
00385         if (x + hw >= sX && y + hh >= sY && ex <= sX + sW && ey <= sY + sH) {
00386             return true;
00387         }
00388 
00389         // Untere bzw. rechte Hälfte selektiert
00390         if (x >= sX && y >= sY && ex - hw <= sX + sW && ey - hh <= sY + sH) {
00391             return true;
00392         }
00393 
00394         return false;
00395     }
