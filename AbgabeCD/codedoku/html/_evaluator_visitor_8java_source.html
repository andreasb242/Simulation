00145     public Object eval(PostfixMathCommandI pfmc, Node children[]) throws ParseException {
00146         if (pfmc instanceof CallbackEvaluationI) {
00147             ASTFunNode node = new ASTFunNode(ParserTreeConstants.JJTFUNNODE);
00148             node.setFunction("TmpFun", pfmc);
00149             node.jjtOpen();
00150             for (int i = 0; i < children.length; ++i)
00151                 node.jjtAddChild(children[i], i);
00152             node.jjtClose();
00153             Object val = ((CallbackEvaluationI) pfmc).evaluate(node, this);
00154             return val;
00155         }
00156 
00157         Stack<Object> lstack = new Stack<Object>();
00158         for (int i = 0; i < children.length; ++i) {
00159             if (!(children[i] instanceof ASTConstant))
00160                 throw new ParseException("buildConstantNode: arguments must all be constant nodes");
00161             lstack.push(((ASTConstant) children[i]).getValue());
00162         }
00163         pfmc.setCurNumberOfParameters(children.length);
00164         pfmc.run(lstack);
00165         return lstack.pop();
00166     }

00190     public Object visit(ASTFunNode node, Object data) throws ParseException {
00191 
00192         if (node == null)
00193             return null;
00194         PostfixMathCommandI pfmc = node.getPFMC();
00195 
00196         // check if the function class is set
00197         if (pfmc == null)
00198             throw new ParseException("No function class associated with " + node.getName());
00199 
00200         if (pfmc instanceof CallbackEvaluationI) {
00201             Object val = ((CallbackEvaluationI) pfmc).evaluate(node, this);
00202             stack.push(val);
00203             return val;
00204         }
00205 
00206         // evaluate all children (each leaves their result on the stack)
00207 
00208         data = node.childrenAccept(this, data);
00209 
00210         if (pfmc.getNumberOfParameters() == -1) {
00211             // need to tell the class how many parameters it can take off
00212             // the stack because it accepts a variable number of params
00213             pfmc.setCurNumberOfParameters(node.jjtGetNumChildren());
00214         }
00215 
00216         // try to run the function
00217 
00218         pfmc.run(stack);
00219 
00220         return data;
00221     }
