00300     public static Node copyChildrenIfNeeded(Node node, Node children[]) throws ParseException {
00301         int n = node.jjtGetNumChildren();
00302         if (n != children.length)
00303             throw new ParseException("copyChildrenIfNeeded: umber of children of node not the same as supplied children");
00304         for (int i = 0; i < n; ++i)
00305             if (node.jjtGetChild(i) != children[i]) {
00306                 node.jjtAddChild(children[i], i);
00307                 children[i].jjtSetParent(node);
00308             }
00309         return node;
00310     }
