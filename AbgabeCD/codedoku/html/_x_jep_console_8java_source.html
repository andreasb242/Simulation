00126             print("Node:\t");

00131             return;

00133             print("Processed:\t");

00136         Node simp = xj.simplify(processed);
00137         print("Simplified:\t");
00138         println(xj.toString(simp));
00139         Object val = xj.evaluate(simp);
00140         String s = xj.getPrintVisitor().formatValue(val);
