00163         synchronized (this) {
00164             if (overridesTimeoutedWaitFinished()) {
00165                 // the the task overrides waitFinished (timeout) or is

00169                     return true;

00173 
00174                 for (;;) {
00175                     wait(milliseconds);

00177                     if (finished) {
00178                         return true;
00179                     }
00180
