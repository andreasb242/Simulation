KeyListener ch.zhaw.simulation.editor.view.AbstractEditorView< C extends AbstractEditorControl<?> >.keyListener [protected]
Initialisierung:
 new KeyAdapter() {

        @Override
        public void keyPressed(KeyEvent e) {
            if (e.getKeyCode() == KeyEvent.VK_ESCAPE) {
                if (showSelection) {
                    showSelection = false;
                    selectionModel.clearTmpSelection();
                    repaint();
                } else {
                    selectionModel.clearSelection();
                }
            } else if (e.getKeyCode() == KeyEvent.VK_DELETE || e.getKeyCode() == KeyEvent.VK_BACK_SPACE) {
                control.deleteSelected();
                e.consume();

            } else {
                ActionListener listener = keyShortcuts.get(e.getKeyChar());
                if (listener != null) {
                    listener.actionPerformed(null);
                }
            }
        }

    }
