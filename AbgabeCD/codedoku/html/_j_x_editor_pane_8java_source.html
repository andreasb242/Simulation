00145      * These next 3 actions are part of a *HACK* to get cut/copy/paste support
00146      * working in the same way as find, undo and redo. in JTextComponent the
00147      * cut/copy/paste actions are _not_ added to the ActionMap. Instead, a
00148      * default "transfer handler" system is used, apparently to get the text
00149      * onto the system clipboard. Since there aren't any CUT/COPY/PASTE actions
00150      * in the JTextComponent's action map, they cannot be referenced by the
00151      * action framework the same way that find/undo/redo are. So, I added the
00152      * actions here. The really hacky part is that by defining an Action to go
00153      * along with the cut/copy/paste keys, I loose the default handling in the
00154      * cut/copy/paste routines. So, I have to remove cut/copy/paste from the
00155      * action map, call the appropriate method (cut, copy, or paste) and then
00156      * add the action back into the map. Yuck!

00308         // JW: fiddling with actionManager's actions state? I'm pretty sure
00309         // we don't want that: the manager will get nuts with multiple
00310         // components with different state.

00528         public int search(Pattern pattern, final int startIndex, boolean backwards) {
00529             if ((pattern == null) || (getDocument().getLength() == 0) || ((startIndex > -1) && (getDocument().getLength() < startIndex))) {
00530                 updateStateAfterNotFound();
00531                 return -1;
00532             }
00533 
00534             int start = startIndex;
00535             if (maybeExtendedMatch(startIndex)) {
00536                 if (foundExtendedMatch(pattern, start)) {
00537                     return lastFoundIndex;
00538                 }
00539                 start++;
00540             }
00541 
00542             int length;
00543             if (backwards) {
00544                 start = 0;
00545                 if (startIndex < 0) {
00546                     length = getDocument().getLength() - 1;
00547                 } else {
00548                     length = -1 + startIndex;
00549                 }
00550             } else {
00551                 // start = startIndex + 1;
00552                 if (start < 0)
00553                     start = 0;
00554                 length = getDocument().getLength() - start;
00555             }
00556             Segment segment = new Segment();
00557 
00558             try {
00559                 getDocument().getText(start, length, segment);
00560             } catch (BadLocationException ex) {
00561                 LOG.log(Level.FINE, "this should not happen (calculated the valid start/length) ", ex);
00562             }
00563 
00564             Matcher matcher = pattern.matcher(segment.toString());
00565             MatchResult currentResult = getMatchResult(matcher, !backwards);
00566             if (currentResult != null) {
00567                 updateStateAfterFound(currentResult, start);
00568             } else {
00569                 updateStateAfterNotFound();
00570             }
00571             return lastFoundIndex;
00572 
00573         }

00586         private boolean foundExtendedMatch(Pattern pattern, int start) {
00587             // JW: logic still needs cleanup...
00588             if (pattern.pattern().equals(lastRegEx)) {
00589                 return false;
00590             }
00591             int length = getDocument().getLength() - start;
00592             Segment segment = new Segment();
00593 
00594             try {
00595                 getDocument().getText(start, length, segment);
00596             } catch (BadLocationException ex) {
00597                 LOG.log(Level.FINE, "this should not happen (calculated the valid start/length) ", ex);
00598             }
00599             Matcher matcher = pattern.matcher(segment.toString());
00600             MatchResult currentResult = getMatchResult(matcher, true);
00601             if (currentResult != null) {
00602                 // JW: how to compare match results reliably?
00603                 // the group().equals probably isn't the best idea...
00604                 // better check pattern?
00605                 if ((currentResult.start() == 0) && (!lastMatchResult.group().equals(currentResult.group()))) {
00606                     updateStateAfterFound(currentResult, start);
00607                     return true;
00608                 }
00609             }
00610             return false;
