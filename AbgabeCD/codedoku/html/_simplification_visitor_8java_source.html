00227             Node newnode = nf.buildOperatorNode(opSet.getSubtract(), lhs, nf.buildConstantNode(opSet.getUMinus(), rhs));

00426     public Node simplifyOp(ASTFunNode node, Node children[]) throws ParseException {
00427         boolean allConst = true;
00428         XOperator op = (XOperator) node.getOperator();
00429         // TODO_YEP a bit of a hack to prevent lists of constants being converted
00430         // what happens is that for [[1,2],[3,4]] the dimension is not passed
00431         // into buildConstantNode so list is treated as [1,2,3,4]
00432         // Ideally there would be a special simplification rule for List
00433         if (op.getPFMC() instanceof org.nfunk.jep.function.List)
00434             return node;
00435         int nchild = children.length;
00436         for (int i = 0; i < nchild; ++i) {
00437             if (!tu.isConstant(children[i]))
00438                 allConst = false;
00439             if (tu.isNaN(children[i]))
00440                 return nf.buildConstantNode(tu.getNAN());
00441         }
00442         if (allConst)
00443             return nf.buildConstantNode(op, children);
00444 
00445         if (nchild == 1) {
00446             if (tu.isUnaryOperator(children[0]) && op == tu.getOperator(children[0])) {
00447                 if (op.isSelfInverse())
00448                     return children[0].jjtGetChild(0);
00449             }
00450         }
00451         if (nchild == 2) {
00452             Node res = null;
00453             if (opSet.getAdd() == op)
00454                 res = simplifyAdd(children[0], children[1]);
00455             if (opSet.getSubtract() == op)
00456                 res = simplifySubtract(children[0], children[1]);
00457             if (opSet.getMultiply() == op)
00458                 res = simplifyMultiply(children[0], children[1]);
00459             if (opSet.getDivide() == op)
00460                 res = simplifyDivide(children[0], children[1]);
00461             if (opSet.getPower() == op)
00462                 res = simplifyPower(children[0], children[1]);
00463             if (res != null) {
00464                 if (tu.isConstant(res))
00465                     return res;
00466                 if (tu.isOperator(res)) {
00467                     Node res2 = simplifyOp((ASTFunNode) res, TreeUtils.getChildrenAsArray(res));
00468                     return res2;
00469                 }
00470                 return res;
00471             }
00472             res = this.simplifyTripple(op, children[0], children[1]);
00473             if (res != null) {
00474                 if (tu.isConstant(res))
00475                     return res;
00476                 if (tu.isOperator(res)) {
00477                     Node res2 = simplifyOp((ASTFunNode) res, TreeUtils.getChildrenAsArray(res));
00478                     return res2;
00479                 }
00480                 return res;
00481             }
00482         }
