00126     public boolean processCommand(String command) {
00127         if (command.equals("quit") || command.equals("exit"))
00128             return false;
00129 
00130         if (command.equals("help")) {
00131             printHelp();
00132             return true;
00133         }
00134 
00135         if (command.equals("functions")) {
00136             printFuns();
00137             return true;
00138         }
00139 
00140         if (command.equals("operators")) {
00141             printOps();
00142             return true;
00143         }
00144 
00145         if (command.equals("variables")) {
00146             printVars();
00147             return true;
00148         }
00149         if (!testSpecialCommands(command))
00150             return true;
00151 
00152         try {
00153             Node n = j.parse(command);
00154             processEquation(n);
00155         } catch (Exception e) {
00156             return handleError(e);
00157         }
00158 
00159         return true;
00160     }

00291     public String[] split(String s) {
00292         StringTokenizer st = new StringTokenizer(s);
00293         int tokCount = st.countTokens();
00294         String res[] = new String[tokCount];
00295         int pos = 0;

00340     public void keyReleased(KeyEvent event) {

00342         if (code == KeyEvent.VK_ENTER) {
00343             int cpos = ta.getCaretPosition();
00344             String alltext = ta.getText();
00345             String before = alltext.substring(0, cpos - 1);
00346             int startOfLine = before.lastIndexOf('\n');
00347             if (startOfLine > 0)
00348                 before = before.substring(startOfLine + 1);
00349             String prompt = getPrompt();
00350             String line = null;
00351             if (before.startsWith(prompt)) {
00352                 line = before.substring(prompt.length());
00353                 this.processCommand(line);
00354             }
00355             // System.out.println("line ("+line+")");
00356             // if(!flag) this.exit();
