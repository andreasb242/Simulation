00033     public void run(Stack<Object> stack) throws ParseException {
00034         checkStack(stack);// check the stack
00035 
00036         Object sum = stack.pop();
00037         Object param;
00038         int i = 1;
00039 
00040         // repeat summation for each one of the current parameters
00041         while (i < curNumberOfParameters) {
00042             // get the parameter from the stack
00043             param = stack.pop();
00044 
00045             // add it to the sum (order is important for String arguments)
00046             sum = add(param, sum);
00047 
00048             i++;
00049         }
00050 
00051         stack.push(sum);
00052 
00053         return;
00054     }

00069     public Object add(Object param1, Object param2) throws ParseException {
00070         if (param1 instanceof Complex) {
00071             if (param2 instanceof Complex)
00072                 return add((Complex) param1, (Complex) param2);
00073             else if (param2 instanceof Number)
00074                 return add((Complex) param1, (Number) param2);
00075         } else if (param1 instanceof Number) {
00076             if (param2 instanceof Complex)
00077                 return add((Complex) param2, (Number) param1);
00078             else if (param2 instanceof Number)
00079                 return add((Number) param1, (Number) param2);
00080         } else if ((param1 instanceof String) && (param2 instanceof String)) {
00081             return (String) param1 + (String) param2;
00082         }
00083 
00084         throw new ParseException("Invalid parameter type");
00085     }
