00218     public void keyPressed(KeyEvent e) {
00219         if (!isVisible()) {
00220             return;
00221         }
00222 
00223         int code = e.getKeyCode();
00224         switch (code) {
00225         case KeyEvent.VK_DOWN:
00226             setSelectNext();
00227             e.consume();
00228             break;
00229         case KeyEvent.VK_UP:
00230             setSelectPrevious();
00231             e.consume();
00232             break;
00233         case KeyEvent.VK_ESCAPE:
00234             setVisible(false);
00235             textField.requestFocus();
00236             e.consume();
00237             break;
00238         }
00239 
00240         if (isCompletionKey(code, textField)) {
00241             File file = (File) list.getSelectedValue();
00242             if (file != null) {
00243                 if (file.equals(chooser.getCurrentDirectory())) {
00244                     chooser.firePropertyChange(JFileChooser.DIRECTORY_CHANGED_PROPERTY, false, true);
00245                 } else {
00246                     chooser.setSelectedFiles(new File[] { file });
00247                     chooser.setCurrentDirectory(file);
00248                 }
00249                 if (file.isDirectory()) {
00250                     try {
00251                         Document doc = textField.getDocument();
00252                         doc.insertString(doc.getLength(), File.separator, null);
00253                     } catch (BadLocationException ex) {
00254                         Logger.getLogger(getClass().getName()).log(Level.FINE, "Cannot append directory separator.", ex);
00255                     }
00256                 }
00257             }
00258             setVisible(false);
00259             textField.requestFocus();
00260             e.consume();
00261         }
00262     }

00272     private boolean isCompletionKey(int keyCode, JTextField textField) {
00273         if (keyCode == KeyEvent.VK_ENTER || keyCode == KeyEvent.VK_TAB) {
00274             return true;
00275         }
00276         if (keyCode == KeyEvent.VK_RIGHT && (textField.getCaretPosition() >= (textField.getDocument().getLength() - 1))) {
00277             return true;
00278         }
00279 
00280         return false;
