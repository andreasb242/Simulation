00307     private void blitSymbol(double x, double y, int size, int symbol, boolean fill) {
00308         Image[][][] images = (Image[][][]) symbols.get(webColor);
00309         if (images == null) {
00310             images = new Image[2][NUMBER_OF_SYMBOLS][MAX_BLIT_SIZE];
00311             symbols.put(webColor, images);
00312         }
00313 
00314         Image image = images[fill ? 1 : 0][symbol][size - 1];
00315         int imageSize = size + 1;
00316         double imageSize2 = imageSize / 2.0;
00317         if (image == null) {
00318             image = getDeviceConfiguration().createCompatibleImage(imageSize + 1, imageSize + 1, Transparency.BITMASK);
00319             VectorGraphics imageGraphics = VectorGraphics.create(image.getGraphics());
00320             Composite composite = imageGraphics.getComposite();
00321             imageGraphics.setComposite(AlphaComposite.Clear);
00322             imageGraphics.fillRect(0, 0, size, size);
00323             imageGraphics.setComposite(composite);
00324             imageGraphics.setColor(getColor());
00325             if (fill) {
00326                 fillSymbol(imageGraphics, imageSize2, imageSize2, size, symbol);
00327             } else {
00328                 drawSymbol(imageGraphics, imageSize2, imageSize2, size, symbol);
00329             }
00330             images[fill ? 1 : 0][symbol][size - 1] = image;
00331         }
00332         drawImage(image, (int) (x - imageSize2), (int) (y - imageSize2), null);
