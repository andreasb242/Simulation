00364     public Vector<XVariable> recursiveGetVarsInEquation(Node n, Vector<XVariable> v) throws ParseException {
00365         if (n instanceof ASTVarNode) {
00366             XVariable var = (XVariable) (((ASTVarNode) n).getVar());
00367             if (!v.contains(var)) {
00368                 if (var.hasEquation())
00369                     recursiveGetVarsInEquation(var.getEquation(), v);
00370                 if (v.contains(var))
00371                     throw new ParseException("Recursive definition for " + var.getName());
00372                 v.add(var);
00373             }
00374         } else if (n instanceof ASTFunNode) {
00375             for (int i = 0; i < n.jjtGetNumChildren(); ++i)
00376                 recursiveGetVarsInEquation(n.jjtGetChild(i), v);
