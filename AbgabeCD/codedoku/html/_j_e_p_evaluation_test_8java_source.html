00178     private Object parseNextLine(BufferedReader reader) throws Exception {
00179         Object value;
00180         String line, errorStr;
00181 
00182         // cycle till a valid line is found
00183         do {
00184             line = reader.readLine(); // returns null on end of file
00185             if (line == null)
00186                 return null;
00187             lineCount++;
00188         } while (line.length() == 0 || line.trim().charAt(0) == '#');
00189 
00190         // parse the expression
00191         myParser.parseExpression(line);
00192         // did an error occur while parsing?
00193         if (myParser.hasError()) {
00194             errorStr = myParser.getErrorInfo();
00195             throw new Exception("Error while parsing line " + lineCount + ": " + errorStr);
00196         }
00197 
00198         // evaluate the expression
00199         value = myParser.getValueAsObject();
00200         // did an error occur while evaluating?
00201         if (value == null || myParser.hasError()) {
00202             errorStr = myParser.getErrorInfo();
00203             throw new Exception("Error while evaluating line " + lineCount + ": " + errorStr);
00204         }
00205 
00206         return value;
00207     }

00226             return Math.abs(((Number) param1).doubleValue() - ((Number) param2).doubleValue()) < tolerance;
