00067     public void print(Node node, PrintStream out) {
00068         sb = new StringBuffer();
00069         acceptCatchingErrors(node, null);
00070         if (maxLen == -1)
00071             out.print(sb);
00072         else {
00073             while (true) {
00074                 if (sb.length() < maxLen) {
00075                     out.print(sb);
00076                     return;
00077                 }
00078                 int pos = maxLen - 2;
00079                 for (int i = maxLen - 2; i >= 0; --i) {
00080                     char c = sb.charAt(i);
00081                     if (c == '+' || c == '-' || c == '*' || c == '/') {
00082                         pos = i;
00083                         break;
00084                     }
00085                 }
00086                 // out.println("<"+sb.substring(0,pos+10)+">");
00087                 out.println(sb.substring(0, pos + 1));
00088                 sb.delete(0, pos + 1);
00089             }
00090         }
00091     }

00190     private boolean testLeft(XOperator top, Node lhs) {
00191         if ((mode & FULL_BRACKET) != 0) {
00192             return true;
00193         } else if (lhs instanceof ASTFunNode && ((ASTFunNode) lhs).isOperator()) {
00194             XOperator lhsop = (XOperator) ((ASTFunNode) lhs).getOperator();
00195             if (top == lhsop) {
00196                 if (top.getBinding() == XOperator.LEFT // (1-2)-3 -> 1-2-3
00197                         && top.isAssociative())
00198                     return false;
00199                 else if (top.useBindingForPrint())
00200                     return false;
00201                 else
00202                     return true; // (1=2)=3 -> (1=2)=3
00203             } else if (top.getPrecedence() == lhsop.getPrecedence()) {
00204                 if (lhsop.getBinding() == XOperator.LEFT && lhsop.isAssociative())
00205                     return false;
00206                 else if (lhsop.useBindingForPrint())
00207                     return false;
00208                 else
00209                     return true;
00210             } // (1=2)=3 -> (1=2)=3
00211 
00212             else if (top.getPrecedence() > lhsop.getPrecedence()) // (1*2)+3

00240         if ((mode & FULL_BRACKET) != 0) {

00279     public Object visit(ASTFunNode node, Object data) throws ParseException {
00280         if (!node.isOperator())
00281             return visitFun(node);
00282         if (node instanceof PrintRulesI) {
00283             ((PrintRulesI) node).append(node, this);
00284             return null;
00285         }
00286         if (node.getOperator() == null) {
00287             throw new ParseException("Null operator in print for " + node);
00288         }
00289         if (specialRules.containsKey(node.getOperator())) {
00290             ((PrintRulesI) specialRules.get(node.getOperator())).append(node, this);
00291             return null;
00292         }
00293         if (node.getPFMC() instanceof org.nfunk.jep.function.List) {
00294             append("[");
00295             for (int i = 0; i < node.jjtGetNumChildren(); ++i) {
00296                 if (i > 0)
00297                     append(",");
00298                 node.jjtGetChild(i).jjtAccept(this, null);
00299             }
00300             append("]");
00301             return null;
00302         }
00303 
00304         if (((XOperator) node.getOperator()).isUnary())
00305             return visitUnary(node, data);
00306 
00307         if (((XOperator) node.getOperator()).isBinary()) {
00308             XOperator top = (XOperator) node.getOperator();
00309             if (node.jjtGetNumChildren() != 2)
00310                 return visitNaryBinary(node, top);
00311             Node lhs = node.jjtGetChild(0);
00312             Node rhs = node.jjtGetChild(1);
00313 
00314             if (testLeft(top, lhs))
00315                 printBrackets(lhs);
00316             else
00317                 printNoBrackets(lhs);
00318 
00319             // now print the node
00320             sb.append(node.getOperator().getSymbol());
00321             // now the rhs
00322 
00323             if (testRight(top, rhs))
00324                 printBrackets(rhs);
00325             else
00326                 printNoBrackets(rhs);
00327 
00328         }
00329         return null;
00330     }
