00033     public boolean test(ASTFunNode node, Node[] children) {
00034         if (!node.isOperator())
00035             return false;
00036         XOperator op = (XOperator) node.getOperator();
00037 
00038         if (opSet.getPower() == op) {
00039             if (tu.getOperator(children[0]) == opSet.getAdd() || tu.getOperator(children[0]) == opSet.getSubtract()) {
00040                 return tu.isInteger(children[1]) && (tu.isPositive(children[1]) || tu.isZero(children[1]));
00041             }
00042             return false;
00043         }

00047     public Node apply(ASTFunNode node, Node[] children) throws ParseException {
00048         Operator lhsOp = tu.getOperator(children[0]);
00049         int n = tu.intValue(children[1]);
00050         Node sub1 = children[0].jjtGetChild(0);
00051         Node sub2 = children[0].jjtGetChild(1);
00052 
00053         // (a+b)^ n --> ( a^n+ nC1 a ^(n-1) b + ....)
00054         if (lhsOp == opSet.getAdd() || lhsOp == opSet.getSubtract()) {
00055             if (n == 0)
00056                 return nf.buildConstantNode(new Double(1));
00057             if (n == 1)
00058                 return children[0];
00059 
00060             Node vals[] = new Node[n + 1];
00061             // a^n
00062             vals[0] = nf.buildOperatorNode(opSet.getPower(), xj.deepCopy(sub1), nf.buildConstantNode(new Double(n)));
00063             if (n == 2) {
00064                 vals[1] = nf.buildOperatorNode(opSet.getMultiply(), nf.buildConstantNode(new Double(2)), nf.buildOperatorNode(opSet.getMultiply(), xj
00065                         .deepCopy(sub1), xj.deepCopy(sub2)));
00066             } else {
00067                 // n * a^(n-1) * b
00068                 vals[1] = nf.buildOperatorNode(opSet.getMultiply(), nf.buildConstantNode(new Double(n)), nf.buildOperatorNode(opSet.getMultiply(), nf
00069                         .buildOperatorNode(opSet.getPower(), xj.deepCopy(sub1), nf.buildConstantNode(new Double(n - 1))), xj.deepCopy(sub2)));
00070             }
00071             // n * a * b^(n-1)
00072             if (n >= 3) {
00073                 vals[n - 1] = nf.buildOperatorNode(opSet.getMultiply(), nf.buildConstantNode(new Double(n)), nf.buildOperatorNode(opSet.getMultiply(), xj
00074                         .deepCopy(sub1), nf.buildOperatorNode(opSet.getPower(), xj.deepCopy(sub2), nf.buildConstantNode(new Double(n - 1)))));
00075             }
00076             // a^n
00077             vals[n] = nf.buildOperatorNode(opSet.getPower(), xj.deepCopy(sub2), nf.buildConstantNode(new Double(n)));
00078             for (int i = 2; i < n - 1; ++i) {
00079                 // (n,i) * a^(n-i) * b^i
00080                 vals[i] = nf.buildOperatorNode(opSet.getMultiply(), nf.buildConstantNode(new Double(XMath.binomial(n, i))), nf.buildOperatorNode(opSet
00081                         .getMultiply(), nf.buildOperatorNode(opSet.getPower(), xj.deepCopy(sub1), nf.buildConstantNode(new Double(n - i))), nf
00082                         .buildOperatorNode(opSet.getPower(), xj.deepCopy(sub2), nf.buildConstantNode(new Double(i)))));
00083             }
00084 
00085             Node sums[] = new Node[n + 1];
00086             sums[n] = vals[n];
00087             for (int i = n - 1; i >= 0; --i) {
00088                 sums[i] = nf.buildOperatorNode(lhsOp, vals[i], sums[i + 1]);
00089             }
00090             return sums[0];
00091         }
00092         throw new ParseException("ExpandBrackets at least one child must be + or -");
