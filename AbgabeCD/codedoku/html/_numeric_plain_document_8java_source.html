00102     public void insertString(int offset, String str, AttributeSet a)
00103             throws BadLocationException {
00104         if (str == null || str.length() == 0) {
00105             return;
00106         }
00107 
00108         Content content = getContent();
00109         int length = content.length();
00110         int originalLength = length;
00111 
00112         parsePos.setIndex(0);
00113 
00114         // Create the result of inserting the new data,
00115         // but ignore the trailing newline
00116         String targetString = content.getString(0, offset) + str
00117                 + content.getString(offset, length - offset - 1);
00118 
00119         // Parse the input string and check for errors
00120         do {
00121             boolean gotPositive = targetString.startsWith(positivePrefix);
00122             boolean gotNegative = targetString.startsWith(negativePrefix);
00123 
00124             length = targetString.length();
00125 
00126             // If we have a valid prefix, the parse fails if the
00127             // suffix is not present and the error is reported
00128             // at index 0. So, we need to add the appropriate
00129             // suffix if it is not present at this point.
00130             if (gotPositive == true || gotNegative == true) {
00131                 String suffix;
00132                 int suffixLength;
00133                 int prefixLength;
00134 
00135                 if (gotPositive == true && gotNegative == true) {
00136                     // This happens if one is the leading part of
00137                     // the other - e.g. if one is "(" and the other "(("
00138                     if (positivePrefixLen > negativePrefixLen) {
00139                         gotNegative = false;
00140                     } else {
00141                         gotPositive = false;
00142                     }
00143                 }
00144 
00145                 if (gotPositive == true) {
00146                     suffix = positiveSuffix;
00147                     suffixLength = positiveSuffixLen;
00148                     prefixLength = positivePrefixLen;
00149                 } else {
00150                     // Must have the negative prefix
00151                     suffix = negativeSuffix;
00152                     suffixLength = negativeSuffixLen;
00153                     prefixLength = negativePrefixLen;
00154                 }
00155 
00156                 // If the string consists of the prefix alone,
00157                 // do nothing, or the result won't parse.
00158                 if (length == prefixLength) {
00159                     break;
00160                 }
00161 
00162                 // We can't just add the suffix, because part of it
00163                 // may already be there. For example, suppose the
00164                 // negative prefix is "(" and the negative suffix is
00165                 // "$)". If the user has typed "(345$", then it is not
00166                 // correct to add "$)". Instead, only the missing part
00167                 // should be added, in this case ")".
00168                 if (targetString.endsWith(suffix) == false) {
00169                     int i;
00170                     for (i = suffixLength - 1; i > 0; i--) {
00171                         if (targetString
00172                                 .regionMatches(length - i, suffix, 0, i)) {
00173                             targetString += suffix.substring(i);
00174                             break;
00175                         }
00176                     }
00177 
00178                     if (i == 0) {
00179                         // None of the suffix was present
00180                         targetString += suffix;
00181                     }
00182 
00183                     length = targetString.length();
00184                 }
00185             }
00186 
00187             format.parse(targetString, parsePos);
00188 
00189             int endIndex = parsePos.getIndex();
00190             if (endIndex == length) {
00191                 break; // Number is acceptable
00192             }
00193 
00194             // Parse ended early
00195             // Since incomplete numbers don't always parse, try
00196             // to work out what went wrong.
00197             // First check for an incomplete positive prefix
00198             if (positivePrefixLen > 0 && endIndex < positivePrefixLen
00199                     && length <= positivePrefixLen
00200                     && targetString.regionMatches(0, positivePrefix, 0, length)) {
00201                 break; // Accept for now
00202             }
00203 
00204             // Next check for an incomplete negative prefix
00205             if (negativePrefixLen > 0 && endIndex < negativePrefixLen
00206                     && length <= negativePrefixLen
00207                     && targetString.regionMatches(0, negativePrefix, 0, length)) {
00208                 break; // Accept for now
00209             }
00210 
00211             // Allow a number that ends with the group
00212             // or decimal separator, if these are in use
00213             char lastChar = targetString.charAt(originalLength - 1);
00214             int decimalIndex = targetString.indexOf(decimalSeparator);
00215             if (format.isGroupingUsed() && lastChar == groupingSeparator
00216                     && decimalIndex == -1) {
00217                 // Allow a "," but only in integer part
00218                 break;
00219             }
00220 
00221             if (format.isParseIntegerOnly() == false
00222                     && lastChar == decimalSeparator
00223                     && decimalIndex == originalLength - 1) {
00224                 // Allow a ".", but only one
00225                 break;
00226             }
00227 
00228             // No more corrections to make: must be an error
00229             if (errorListener != null) {
00230                 errorListener.insertFailed(this, offset, str, a);
00231             }
00232             return;
00233         } while (true == false);
00234 
00235         // Finally, add to the model
00236         super.insertString(offset, str, a);
00237     }
