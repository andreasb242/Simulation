00049     public MatrixNodeI preprocess(ASTFunNode node, MatrixPreprocessor visitor, MatrixJep jep, MatrixNodeFactory nf) throws ParseException {
00050         MatrixNodeI children[] = visitor.visitChildrenAsArray(node, null);
00051         Operator listOp = ((MatrixOperatorSet) jep.getOperatorSet()).getMList();
00052         // What if we have x=[1,2]; y = [x,x]; or z=[[1,2],x];
00053         // first check if all arguments are TENSORS
00054         boolean flag = true;
00055         if (children[0] instanceof ASTMFunNode) {
00056             if (((ASTMFunNode) children[0]).getOperator() != listOp) {
00057                 flag = false;
00058             }
00059         } else
00060             flag = false;
00061 
00062         if (flag) {
00063             ASTMFunNode opNode1 = (ASTMFunNode) children[0];
00064             Dimensions dim = Dimensions.valueOf(children.length, opNode1.getDim());
00065             ASTMFunNode res = (ASTMFunNode) nf.buildUnfinishedOperatorNode(listOp);
00066             int k = 0;
00067             res.setDim(dim);
00068             res.jjtOpen();
00069             for (int i = 0; i < children.length; ++i) {
00070                 ASTMFunNode opNode = (ASTMFunNode) children[i];
00071                 for (int j = 0; j < opNode.jjtGetNumChildren(); ++j) {
00072                     Node child = opNode.jjtGetChild(j);
00073                     res.jjtAddChild(child, k++);
00074                     child.jjtSetParent(res);
00075                 }
00076             }
00077             res.jjtClose();
00078             return res;
00079         }
00080         MatrixNodeI node1 = children[0];
00081         Dimensions dim = Dimensions.valueOf(children.length, node1.getDim());
00082         ASTMFunNode res = (ASTMFunNode) nf.buildOperatorNode(listOp, children, dim);
00083         return res;
00084     }

00089     private void bufferAppend(MatrixNodeI node, PrintVisitor pv, int currank) throws ParseException {
00090         pv.append("[");
00091         if (currank + 1 >= node.getDim().rank()) {
00092             // bottom of tree
00093             for (int i = 0; i < node.getDim().getIthDim(currank); ++i) {
00094                 if (i != 0)
00095                     pv.append(",");
00096                 node.jjtGetChild(curEle++).jjtAccept(pv, null);
00097             }
00098         } else {
00099             // not bottom of tree
00100             for (int i = 0; i < node.getDim().getIthDim(currank); ++i) {
00101                 if (i != 0)
00102                     pv.append(",");
00103                 bufferAppend(node, pv, currank + 1);
00104             }
00105         }
00106         pv.append("]");
00107     }
