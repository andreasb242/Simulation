00550     public Node parseExpression(String expression_in) {
00551         Reader reader = new StringReader(expression_in);
00552 
00553         try {
00554             // try parsing
00555             errorList.removeAllElements();
00556             topNode = parser.parseStream(reader, this);
00557 
00558             // if there is an error in the list, the parse failed
00559             // so set topNode to null
00560             if (hasError())
00561                 topNode = null;
00562         } catch (Throwable e) {
00563             // an exception was thrown, so there is no parse tree
00564             topNode = null;
00565 
00566             // check the type of error
00567             if (e instanceof ParseException) {
00568                 // the ParseException object contains additional error
00569                 // information
00570                 errorList.addElement(((ParseException) e).getMessage());
00571                 // getErrorInfo());
00572             } else {
00573                 // if the exception was not a ParseException, it was most
00574                 // likely a syntax error
00575                 if (debug) {
00576                     System.out.println(e.getMessage());
00577                     e.printStackTrace();
00578                 }
00579                 errorList.addElement("Syntax error");
00580             }
00581         }
00582 
00583         // If traversing is enabled, print a dump of the tree to
00584         // standard output
00585         if (traverse && !hasError()) {
00586             ParserVisitor v = new ParserDumpVisitor();
00587             try {
00588                 topNode.jjtAccept(v, null);
00589             } catch (ParseException e) {
00590                 errorList.addElement(e.getMessage());
00591             }
00592         }
00593 
00594         return topNode;
00595     }

00648     public double getValue() {

00695     public Object getValueAsObject() {
00696         Object result;
00697 
00698         // fail by returning null if no topNode is available
00699         if (topNode == null)
00700             return null;
00701 
00702         // evaluate the expression
00703         try {
00704             result = ev.getValue(topNode, symTab);
00705         } catch (ParseException e) {
00706             if (debug)
00707                 System.out.println(e);
00708             errorList.addElement("Error during evaluation: " + e.getMessage());
00709             return null;
00710         } catch (RuntimeException e) {
00711             if (debug)
00712                 System.out.println(e);
00713             errorList.addElement(e.getClass().getName() + ": " + e.getMessage());
00714             return null;
00715         }
00716         return result;
00717     }
