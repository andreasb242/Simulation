00403     public Complex power(double exponent) {
00404         // z^exp = abs(z)^exp * (cos(exp*arg(z)) + i*sin(exp*arg(z)))
00405         double scalar = Math.pow(abs(), exponent);
00406         boolean specialCase = false;
00407         int factor = 0;
00408 
00409         // consider special cases to avoid floating point errors
00410         // for power expressions such as (-1e20)^2
00411         if (im == 0 && re < 0) {
00412             specialCase = true;
00413             factor = 2;
00414         }
00415         if (re == 0 && im > 0) {
00416             specialCase = true;
00417             factor = 1;
00418         }
00419         if (re == 0 && im < 0) {
00420             specialCase = true;
00421             factor = -1;
00422         }
00423 
00424         if (specialCase && factor * exponent == (int) (factor * exponent)) {
00425             short[] cSin = { 0, 1, 0, -1 }; // sin of 0, pi/2, pi, 3pi/2
00426             short[] cCos = { 1, 0, -1, 0 }; // cos of 0, pi/2, pi, 3pi/2
00427 
00428             int x = ((int) (factor * exponent)) % 4;
00429             if (x < 0)
00430                 x = 4 + x;
00431 
00432             return new Complex(scalar * cCos[x], scalar * cSin[x]);
00433         }
00434 
00435         double temp = exponent * arg();
00436 
00437         return new Complex(scalar * Math.cos(temp), scalar * Math.sin(temp));
00438     }
