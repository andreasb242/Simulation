00830     private class ImageHandler implements ImageObserver {
00831         // This can come on any thread. If we are in the process of reloading
00832         // the image and determining our state (loading == true) we don't fire
00833         // preference changed, or repaint, we just reset the fWidth/fHeight as
00834         // necessary and return. This is ok as we know when loading finishes
00835         // it will pick up the new height/width, if necessary.
00836         public boolean imageUpdate(Image img, int flags, int x, int y, int newWidth, int newHeight) {
00837             if (image == null || image != img || getParent() == null) {
00838                 return false;
00839             }
00840 
00841             // Bail out if there was an error:
00842             if ((flags & (ABORT | ERROR)) != 0) {
00843                 repaint(0);
00844                 synchronized (InternImageView.this) {
00845                     if (image == img) {
00846                         // Be sure image hasn't changed since we don't
00847                         // initialy synchronize
00848                         image = null;
00849                         if ((state & WIDTH_FLAG) != WIDTH_FLAG) {
00850                             width = DEFAULT_WIDTH;
00851                         }
00852                         if ((state & HEIGHT_FLAG) != HEIGHT_FLAG) {
00853                             height = DEFAULT_HEIGHT;
00854                         }
00855                     }
00856                     if ((state & LOADING_FLAG) == LOADING_FLAG) {
00857                         // No need to resize or repaint, still in the process
00858                         // of loading.
00859                         return false;
00860                     }
00861                 }
00862                 updateAltTextView();
00863                 safePreferenceChanged();
00864                 return false;
00865             }
00866 
00867             // Resize image if necessary:
00868             short changed = 0;
00869             if ((flags & ImageObserver.HEIGHT) != 0 && !getElement().getAttributes().isDefined(HTML.Attribute.HEIGHT)) {
00870                 changed |= 1;
00871             }
00872             if ((flags & ImageObserver.WIDTH) != 0 && !getElement().getAttributes().isDefined(HTML.Attribute.WIDTH)) {
00873                 changed |= 2;
00874             }
00875 
00876             synchronized (InternImageView.this) {
00877                 if (image != img) {
00878                     return false;
00879                 }
00880                 if ((changed & 1) == 1 && (state & WIDTH_FLAG) == 0) {
00881                     width = newWidth;
00882                 }
00883                 if ((changed & 2) == 2 && (state & HEIGHT_FLAG) == 0) {
00884                     height = newHeight;
00885                 }
00886                 if ((state & LOADING_FLAG) == LOADING_FLAG) {
00887                     // No need to resize or repaint, still in the process of
00888                     // loading.
00889                     return true;
00890                 }
00891             }
00892             if (changed != 0) {
