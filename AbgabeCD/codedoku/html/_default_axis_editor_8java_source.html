00347     public void actionPerformed(ActionEvent event) {
00348         String command = event.getActionCommand();
00349         if (command.equals("SelectLabelFont")) {
00350             attemptLabelFontSelection();
00351         } else if (command.equals("SelectLabelPaint")) {
00352             attemptModifyLabelPaint();
00353         } else if (command.equals("SelectTickLabelFont")) {
00354             attemptTickLabelFontSelection();
00355         }
00356     }

00362         Font font = FontChooser.chooseFont(this, this.labelFont);
00363 
00364         if (font != null) {
00365             this.labelFont = font;
00366             this.labelFontField.setText(this.labelFont.getFontName() + " " + this.labelFont.getSize());
00367         }
