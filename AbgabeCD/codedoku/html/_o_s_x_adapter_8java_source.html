00090     public static void setHandler(OSXAdapter adapter) {
00091         try {
00092             Class<?> applicationClass = Class.forName("com.apple.eawt.Application");
00093             if (macOSXApplication == null) {
00094                 macOSXApplication = applicationClass.getConstructor((Class[]) null).newInstance((Object[]) null);
00095             }
00096             Class<?> applicationListenerClass = Class.forName("com.apple.eawt.ApplicationListener");
00097             Method addListenerMethod = applicationClass.getDeclaredMethod("addApplicationListener", new Class[] { applicationListenerClass });
00098             // Create a proxy object around this handler that can be
00099             // reflectively added as an Apple ApplicationListener
00100             Object osxAdapterProxy = Proxy.newProxyInstance(OSXAdapter.class.getClassLoader(), new Class[] { applicationListenerClass }, adapter);
00101             addListenerMethod.invoke(macOSXApplication, new Object[] { osxAdapterProxy });
00102         } catch (ClassNotFoundException e) {
00103             Errorhandler.logError(e, "This version of Mac OS X does not support the Apple EAWT.  ApplicationEvent handling has been disabled (" + e + ")");

00105             // IllegalAccessException loading/invoking
00106             // eawt.Application methods
00107             Errorhandler.logError(e, "Mac OS X Adapter could not talk to EAWT:");
00108         }

00132     // InvocationHandler implementation
00133     // This is the entry point for our proxy object; it is called every time an
00134     // ApplicationListener method is invoked
00135     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
00136         if (isCorrectMethod(method, args)) {
00137             boolean handled = callTarget(args[0]);
00138             setApplicationEventHandled(args[0], handled);
00139         }
00140         // All of the ApplicationListener methods are void; return null
00141         // regardless of what happens
00142         return null;
00143     }

00145     // Compare the method that was called to the intended method when the
00146     // OSXAdapter instance was created
00147     // (e.g. handleAbout, handleQuit, handleOpenFile, etc.)
00148     protected boolean isCorrectMethod(Method method, Object[] args) {
00149         return (targetMethod != null && proxySignature.equals(method.getName()) && args.length == 1);
00150     }

00152     // It is important to mark the ApplicationEvent as handled and cancel the
00153     // default behavior
00154     // This method checks for a boolean result from the proxy method and sets
00155     // the event accordingly
00156     protected void setApplicationEventHandled(Object event, boolean handled) {
00157         if (event != null) {
00158             try {
00159                 Method setHandledMethod = event.getClass().getDeclaredMethod("setHandled", new Class[] { boolean.class });
00160                 // If the target method returns a boolean, use that as a hint
00161                 setHandledMethod.invoke(event, new Object[] { Boolean.valueOf(handled) });
00162             } catch (Exception e) {
00163                 Errorhandler.logError(e, "OSXAdapter was unable to handle an ApplicationEvent: " + event);
00164             }
00165         }
