00039         // evaluate the value of the righthand side.

00041 
00042         // Set the value of the variable on the lhs.
00043         Node lhsNode = node.jjtGetChild(0);
00044         if (lhsNode instanceof ASTVarNode) {
00045             ASTVarNode vn = (ASTVarNode) lhsNode;
00046             Variable var = vn.getVar();
00047             var.setValue(rhsVal);
00048             return rhsVal;
00049         } else if (lhsNode instanceof ASTFunNode && ((ASTFunNode) lhsNode).getPFMC() instanceof LValueI) {
00050             ((LValueI) ((ASTFunNode) lhsNode).getPFMC()).set(pv, lhsNode, rhsVal);
00051             return rhsVal;
00052         }
00053         throw new ParseException("Assignment should have a variable or LValue for the lhs.");
