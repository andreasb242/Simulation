00043         for (int i = 0; i < length; ++i) {
00044             if (terms[i].equals(term)) {

00049         // insert in correct posn
00050         PNodeI newTerms[] = new PNodeI[length + 1];
00051         PNodeI newPowers[] = new PNodeI[length + 1];

00053         boolean done = false;
00054         for (int i = 0; i < length; ++i) {
00055             if (!done && terms[i].compareTo(term) > 0) {
00056                 newTerms[pos] = term;
00057                 newPowers[pos] = power;

00060             }
00061             newTerms[pos] = terms[i];

00083         int numZeros = 0;

00092                 ++numConst;

00103             } // x^0 --> 1

00105             } // 1^x --> 1
